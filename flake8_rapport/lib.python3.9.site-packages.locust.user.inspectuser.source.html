28 def _calc_distribution(user_classes, num_users):
29 fixed_count = sum(u.fixed_count for u in user_classes if u.fixed_count)
30 total_weight = sum(u.weight for u in user_classes if not u.fixed_count)
31 num_users = num_users or (total_weight if not fixed_count else 1)
32 weighted_count = num_users - fixed_count
33 weighted_count = weighted_count if weighted_count > 0 else 0
34 user_classes_count = {}
35

37 count = u.fixed_count if u.fixed_count else (u.weight / total_weight) * weighted_count
