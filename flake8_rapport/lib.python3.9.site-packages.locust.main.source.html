426 def shutdown():
427 """
428 Shut down locust by firing quitting event, printing/writing stats and exiting
429 """
430 logger.debug("Running teardowns...")
431  
432 if input_listener_greenlet is not None:
433 input_listener_greenlet.kill(block=False)
434  
435 environment.events.quitting.fire(environment=environment, reverse=True)
436  
437 # determine the process exit code
438 if environment.process_exit_code is not None:
439 code = environment.process_exit_code
440 elif len(runner.errors) or len(runner.exceptions):
441 code = options.exit_code_on_error
442 elif log.unhandled_greenlet_exception:
443 code = 2
444 else:
445 code = 0
446  
447 logger.info(f"Shutting down (exit code {code})")
448  
449 if stats_printer_greenlet is not None:
450 stats_printer_greenlet.kill(block=False)
451 if headless_master_greenlet is not None:
452 headless_master_greenlet.kill(block=False)
453 logger.debug("Cleaning up runner...")
454 if runner is not None:
455 runner.quit()
456 if options.json:
457 print_stats_json(runner.stats)
458 elif not isinstance(runner, locust.runners.WorkerRunner):
459 print_stats(runner.stats, current=False)
460 print_percentile_stats(runner.stats)
461 print_error_report(runner.stats)
462  
463 environment.events.quit.fire(exit_code=code)
464 sys.exit(code)
