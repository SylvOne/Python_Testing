96 def test_specify_config_file(self):

98 textwrap.dedent(
99 """
100 host = localhost # With "="
101 u 100 # Short form
102 spawn-rate 5 # long form
103 # boolean
104 headless
105 # (for some reason an inline comment makes boolean values fail in configargparse nowadays)
106 """
107 ),
108 suffix=".conf",
109 ) as conf_file_path:
110 options = parse_options(
111 args=[
112 "--config",
113 conf_file_path,
114 ]
115 )
116 self.assertEqual(conf_file_path, options.config)
117 self.assertEqual("localhost", options.host)
118 self.assertEqual(100, options.num_users)
