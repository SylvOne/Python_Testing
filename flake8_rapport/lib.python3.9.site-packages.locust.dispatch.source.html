29 class UsersDispatcher(Iterator):
30 """
31 Iterator that dispatches the users to the workers.
32  
33 The dispatcher waits an appropriate amount of time between each iteration
34 in order for the spawn rate to be respected whether running in
35 local or distributed mode.
36  
37 The terminology used in the users dispatcher is:
38 - Dispatch cycle
39 A dispatch cycle corresponds to a ramp-up from start to finish. So,
40 going from 10 to 100 users with a spawn rate of 1/s corresponds to one
41 dispatch cycle. An instance of the `UsersDispatcher` class "lives" for
42 one dispatch cycle only.
43 - Dispatch iteration
44 A dispatch cycle contains one or more dispatch iterations. In the previous example
45 of going from 10 to 100 users with a spawn rate of 1/s, there are 100 dispatch iterations.
46 That is, from 10 to 11 users is a dispatch iteration, from 12 to 13 is another, and so on.
47 If the spawn rate were to be 2/s, then there would be 50 dispatch iterations for this dispatch cycle.
48 For a more extreme case with a spawn rate of 120/s, there would be only a single dispatch iteration
49 from 10 to 100.
50 """

120 def _sort_workers(self):
121 # Sorting workers ensures repeatable behaviour
122 worker_nodes_by_id = sorted(self._worker_nodes, key=lambda w: w.id)
123  
124 # Give every worker an index indicating how many workers came before it on that host
125 workers_per_host = defaultdict(lambda: 0)
126 for worker_node in worker_nodes_by_id:
127 host = worker_node.id.split("_")[0]
128 worker_node._index_within_host = workers_per_host[host]
129 workers_per_host[host] = workers_per_host[host] + 1
130  
131 # Sort again, first by index within host, to ensure Users get started evenly across hosts
132 self._worker_nodes = sorted(self._worker_nodes, key=lambda worker: (worker._index_within_host, worker.id))

224 def _prepare_rebalance(self) -> None:
225 """
226 When a rebalance is required because of added and/or removed workers, we compute the desired state as if
227 we started from 0 user. So, if we were currently running 500 users, then the `_distribute_users` will
228 perform a fake ramp-up without any waiting and return the final distribution.
229 """

231 # fixed users to add.

271 def _add_users_on_workers(self) -> Dict[str, Dict[str, int]]:
272 """Add users on the workers until the target number of users is reached for the current dispatch iteration
273  
274 :return: The users that we want to run on the workers
275 """
276 current_user_count_target = min(
277 self._current_user_count + self._user_count_per_dispatch_iteration, self._target_user_count
278 )

281 if not user:
282 self._no_user_to_spawn = True
283 break
284 worker_node = next(self._worker_node_generator)
285 self._users_on_workers[worker_node.id][user] += 1
286 self._current_user_count += 1
287 self._active_users.append((worker_node, user))
288 if self._current_user_count >= current_user_count_target:

293 def _remove_users_from_workers(self) -> Dict[str, Dict[str, int]]:
294 """Remove users from the workers until the target number of users is reached for the current dispatch iteration

299 self._current_user_count - self._user_count_per_dispatch_iteration, self._target_user_count

352 def _user_gen(self) -> Generator[Optional[str], None, None]:
353 """
354 This method generates users according to their weights using
355 a smooth weighted round-robin algorithm implemented by https://github.com/linnik/roundrobin.
356  
357 For example, given users A, B with weights 5 and 1 respectively, this algorithm
358 will yield AAABAAAAABAA. The smooth aspect of this algorithm is what makes it possible
359 to keep the distribution during ramp-up and ramp-down. If we were to use a normal
360 weighted round-robin algorithm, we'd get AAAAABAAAAAB which would make the distribution
361 less accurate during ramp-up/down.
362 """
363  
364 def infinite_cycle_gen(users: List[Tuple[Type[User], int]]) -> itertools.cycle:
365 if not users:
366 return itertools.cycle([None])
367  
368 # Normalize the weights so that the smallest weight will be equal to "target_min_weight".
369 # The value "2" was experimentally determined because it gave a better distribution especially
370 # when dealing with weights which are close to each others, e.g. 1.5, 2, 2.4, etc.
371 target_min_weight = 2
372  
373 # 'Value' here means weight or fixed count
374 normalized_values = [
375 (
376 user.__name__,
377 round(target_min_weight * value / min(u[1] for u in users)),
378 )
379 for user, value in users
380 ]
381 generation_length_to_get_proper_distribution = sum(
382 normalized_val[1] for normalized_val in normalized_values
383 )
384 gen = smooth(normalized_values)
385  
386 # Instead of calling `gen()` for each user, we cycle through a generator of fixed-length
387 # `generation_length_to_get_proper_distribution`. Doing so greatly improves performance because
388 # we only ever need to call `gen()` a relatively small number of times. The length of this generator
389 # is chosen as the sum of the normalized weights. So, for users A, B, C of weights 2, 5, 6, the length is
390 # 2 + 5 + 6 = 13 which would yield the distribution `CBACBCBCBCABC` that gets repeated over and over
391 # until the target user count is reached.
392 return itertools.cycle(gen() for _ in range(generation_length_to_get_proper_distribution))
393  
394 fixed_users = {u.__name__: u for u in self._user_classes if u.fixed_count}
395  
396 cycle_fixed_gen = infinite_cycle_gen([(u, u.fixed_count) for u in fixed_users.values()])
397 cycle_weighted_gen = infinite_cycle_gen([(u, u.weight) for u in self._user_classes if not u.fixed_count])
398  
399 # Spawn users
400 while True:
401 if self._try_dispatch_fixed:
402 self._try_dispatch_fixed = False
403 current_fixed_users_count = {u: self._get_user_current_count(u) for u in fixed_users}
404 spawned_classes: Set[str] = set()
405 while len(spawned_classes) != len(fixed_users):
406 user_name: Optional[str] = next(cycle_fixed_gen)
407 if not user_name:
408 break
409  
410 if current_fixed_users_count[user_name] < fixed_users[user_name].fixed_count:
411 current_fixed_users_count[user_name] += 1
412 if current_fixed_users_count[user_name] == fixed_users[user_name].fixed_count:
413 spawned_classes.add(user_name)
414 yield user_name
415  
416 # 'self._try_dispatch_fixed' was changed outhere, we have to recalculate current count
417 if self._try_dispatch_fixed:
418 current_fixed_users_count = {u: self._get_user_current_count(u) for u in fixed_users}
419 spawned_classes.clear()
420 self._try_dispatch_fixed = False
421  
422 yield next(cycle_weighted_gen)
