223 def _remove_user_classes_with_weight_zero(self) -> None:
224 """
225 Remove user classes having a weight of zero.
226 """
227 if len(self.user_classes) == 0:
228 # Preserve previous behaviour that allowed no user classes to be specified.
229 return
230 filtered_user_classes = [
231 user_class for user_class in self.user_classes if user_class.weight > 0 or user_class.fixed_count > 0
232 ]
233 if len(filtered_user_classes) == 0:

238 def assign_equal_weights(self) -> None:
239 """
240 Update the user classes such that each user runs their specified tasks with equal
241 probability.
242 """
243 for u in self.user_classes:
244 u.weight = 1
245 user_tasks: List[TaskSet | Callable] = []
246 tasks_frontier = u.tasks
247 while len(tasks_frontier) != 0:
248 t = tasks_frontier.pop()
249 if isinstance(t, TaskHolder):
250 tasks_frontier.extend(t.tasks)
251 elif callable(t):
252 if t not in user_tasks:
253 user_tasks.append(t)
254 else:
255 raise ValueError("Unrecognized task type in user")
256 u.tasks = user_tasks
