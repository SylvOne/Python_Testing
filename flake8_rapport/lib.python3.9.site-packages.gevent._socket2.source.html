241 def makefile(self, mode='r', bufsize=-1):
242 # Two things to look out for:
243 # 1) Closing the original socket object should not close the
244 # fileobject (hence creating a new socket instance);
245 # An alternate approach is what _socket3.py does, which is to
246 # keep count of the times makefile objects have been opened (Py3's
247 # SocketIO helps with that). But the newly created socket, which
248 # has its own read/write watchers, does need those to be closed
249 # when the fileobject is; our custom subclass does that. Note that
250 # we can't pass the 'close=True' argument, as that causes reference counts
251 # to get screwed up, and Python2 sockets rely on those.
252 # 2) The resulting fileobject must keep the timeout in order
253 # to be compatible with the stdlib's socket.makefile.
254 # Pass self as _sock to preserve timeout.
255 fobj = _fileobject(type(self)(_sock=self), mode, bufsize)
256 if PYPY:
257 self._sock._drop()
258 return fobj
